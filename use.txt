	@GetMapping(value="/PersonalDetails/{id}")
		public ResponseEntity<PersonalDetails> startPageCS(@PathVariable int id) 
		{
			Optional<PersonalDetails> sop=hs.getSingle(id);
			if(sop.isPresent()) {
			PersonalDetails sret=sop.get();
			return new ResponseEntity<>(sret,HttpStatus.OK);
			}
			else
			{
				throw new CustomerNotFoundException("Customer Not Avialable");
			}
			
		}
